import React, { useState, useRef, useEffect } from 'react';


// 載入外部函式庫用於匯出功能

const loadScript = (src) => {

  return new Promise((resolve, reject) => {

    const script = document.createElement('script');

    script.src = src;

    script.onload = resolve;

    script.onerror = reject;

    document.head.appendChild(script);

  });

};


const App = () => {

  // 建立 9x9 的資料結構 (81格)

  // data[blockIndex][cellIndex]

  // blockIndex: 0-8 代表九個大區塊，cellIndex: 0-8 代表區塊內的九個小格

  const [data, setData] = useState(Array(9).fill(null).map(() => Array(9).fill('')));

  const [isExporting, setIsExporting] = useState(false);

  const gridRef = useRef(null);


  useEffect(() => {

    // 預載入匯出所需的外掛程式

    Promise.all([

      loadScript('https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js'),

      loadScript('https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js')

    ]).catch(err => console.error("無法載入匯出元件", err));

  }, []);


  // 核心連動邏輯

  const handleCellChange = (blockIdx, cellIdx, value) => {

    const newData = data.map(block => [...block]);

    newData[blockIdx][cellIdx] = value;


    // 當修改中央區塊 (Index 4) 的非中心格時，同步更新對應周圍區塊的中心點

    if (blockIdx === 4 && cellIdx !== 4) {

      newData[cellIdx][4] = value;

    } 

    // 當修改周圍區塊 (Index != 4) 的中心點 (Index 4) 時，同步回報給中央區塊

    else if (blockIdx !== 4 && cellIdx === 4) {

      newData[4][blockIdx] = value;

    }


    setData(newData);

  };


  const exportAsImage = async () => {

    if (!window.html2canvas) {

      console.error("html2canvas 未載入");

      return;

    }

    setIsExporting(true);

    // 延遲以確保輸入框失焦，避免出現游標或藍色框框

    setTimeout(async () => {

      const canvas = await window.html2canvas(gridRef.current, {

        backgroundColor: '#ffffff',

        scale: 2,

        useCORS: true,

      });

      const link = document.createElement('a');

      link.download = `目標九宮格-${data[4][4] || '我的目標'}.png`;

      link.href = canvas.toDataURL('image/png');

      link.click();

      setIsExporting(false);

    }, 150);

  };


  const exportAsExcel = () => {

    if (!window.XLSX) {

      console.error("xlsx 未載入");

      return;

    }

    

    // 建立 9x9 的試算表內容

    const rows = [];

    for (let rowBlock = 0; rowBlock < 3; rowBlock++) {

      for (let rowCell = 0; rowCell < 3; rowCell++) {

        const rowData = [];

        for (let colBlock = 0; colBlock < 3; colBlock++) {

          const blockIdx = rowBlock * 3 + colBlock;

          for (let colCell = 0; colCell < 3; colCell++) {

            const cellIdx = rowCell * 3 + colCell;

            rowData.push(data[blockIdx][cellIdx]);

          }

        }

        rows.push(rowData);

      }

    }


    const worksheet = window.XLSX.utils.aoa_to_sheet(rows);

    const workbook = window.XLSX.utils.book_new();

    window.XLSX.utils.book_append_sheet(workbook, worksheet, "目標九宮格");

    window.XLSX.writeFile(workbook, `目標九宮格-${data[4][4] || '我的目標'}.xlsx`);

  };


  const resetGrid = () => {

    if (window.confirm("確定要清除所有填寫內容嗎？")) {

      setData(Array(9).fill(null).map(() => Array(9).fill('')));

    }

  };


  return (

    <div className="min-h-screen bg-slate-50 py-8 px-4 font-sans text-slate-800">

      <div className="max-w-[1000px] mx-auto">

        {/* 標題與操作區 */}

        <div className="flex flex-col md:flex-row md:items-end justify-between mb-8 gap-4">

          <div>

            <h1 className="text-3xl font-extrabold text-slate-900 mb-2">目標九宮格</h1>

            <p className="text-slate-500">從中心點出發，系統會自動同步您的關鍵要素至對應區塊。</p>

          </div>

          <div className="flex gap-2">

            <button 

              onClick={exportAsImage} 

              className="px-4 py-2 bg-indigo-600 hover:bg-indigo-700 text-white rounded-lg shadow-sm transition-all flex items-center gap-2"

            >

              <svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><rect width="18" height="18" x="3" y="3" rx="2" ry="2"/><circle cx="9" cy="9" r="2"/><path d="m21 15-3.086-3.086a2 2 0 0 0-2.828 0L6 21"/></svg>

              匯出圖片

            </button>

            <button 

              onClick={exportAsExcel} 

              className="px-4 py-2 bg-emerald-600 hover:bg-emerald-700 text-white rounded-lg shadow-sm transition-all flex items-center gap-2"

            >

              <svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M14.5 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V7.5L14.5 2z"/><polyline points="14 2 14 8 20 8"/><line x1="16" x2="8" y1="13" y2="13"/><line x1="16" x2="8" y1="17" y2="17"/></svg>

              匯出表格

            </button>

            <button 

              onClick={resetGrid} 

              className="px-4 py-2 bg-white border border-slate-200 text-slate-500 hover:bg-red-50 hover:text-red-600 rounded-lg shadow-sm transition-all"

            >

              清除全部

            </button>

          </div>

        </div>


        {/* 81格 容器 */}

        <div className="bg-white p-4 rounded-2xl shadow-xl overflow-x-auto">

          <div 

            ref={gridRef}

            className={`grid grid-cols-3 gap-2 bg-slate-300 p-2 border-2 border-slate-300 ${isExporting ? 'w-[900px] h-[900px]' : 'w-full aspect-square min-w-[600px]'}`}

          >

            {data.map((block, bIdx) => (

              <div 

                key={bIdx} 

                className={`grid grid-cols-3 gap-0.5 p-1 rounded-sm

                  ${bIdx === 4 ? 'bg-indigo-50 border-2 border-indigo-200' : 'bg-slate-100'}`}

              >

                {block.map((cellValue, cIdx) => {

                  const isMainGoal = bIdx === 4 && cIdx === 4;

                  const isSubGoalTrigger = bIdx === 4 && cIdx !== 4;

                  const isSyncCenter = bIdx !== 4 && cIdx === 4;


                  let cellClass = "bg-white text-slate-700";

                  if (isMainGoal) cellClass = "bg-indigo-600 text-white font-bold placeholder-indigo-300";

                  else if (isSubGoalTrigger) cellClass = "bg-indigo-100 text-indigo-900 font-semibold";

                  else if (isSyncCenter) cellClass = "bg-slate-200 text-slate-900 font-semibold";


                  return (

                    <div 

                      key={cIdx} 

                      className={`relative flex items-center justify-center border border-slate-200 group transition-all h-full ${cellClass}`}

                    >

                      <textarea

                        className="w-full h-full bg-transparent resize-none focus:outline-none p-1 text-center text-[10px] sm:text-xs leading-tight flex items-center justify-center border-none"

                        value={cellValue}

                        placeholder={isMainGoal ? "目標" : ""}

                        onChange={(e) => handleCellChange(bIdx, cIdx, e.target.value)}

                        style={{ 

                          textAlignLast: 'center',

                          paddingTop: 'auto',

                          paddingBottom: 'auto',

                          verticalAlign: 'middle'

                        }}

                      />

                    </div>

                  );

                })}

              </div>

            ))}

          </div>

        </div>


        {/* 操作說明卡片 */}

        <div className="mt-8 grid grid-cols-1 md:grid-cols-3 gap-4">

          <div className="p-4 bg-white rounded-xl border border-slate-100 shadow-sm">

            <h4 className="font-bold text-indigo-600 mb-2">1. 設定核心</h4>

            <p className="text-sm text-slate-500">在正中央深藍色的格子內填寫您的主要目標。</p>

          </div>

          <div className="p-4 bg-white rounded-xl border border-slate-100 shadow-sm">

            <h4 className="font-bold text-indigo-600 mb-2">2. 拆解要素</h4>

            <p className="text-sm text-slate-500">在中央區塊的其他 8 格填寫關鍵要素，系統會自動同步至周圍。</p>

          </div>

          <div className="p-4 bg-white rounded-xl border border-slate-100 shadow-sm">

            <h4 className="font-bold text-indigo-600 mb-2">3. 具體行動</h4>

            <p className="text-sm text-slate-500">在周圍區塊的空白格內填入具體執行計畫，完成 81 格思考。</p>

          </div>

        </div>

      </div>


      <footer className="text-center py-10 text-slate-400 text-xs">

        目標九宮格思考工具 · 支援自動同步與多格式匯出

      </footer>

    </div>

  );

};


export default App;
